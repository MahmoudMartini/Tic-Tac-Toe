    game.update_square_state(i, j)
    
    game.update_game_state()
    if game.get_game_state() == 'Draw':
        message_label.text = "Draw"
        for button in square_buttons: button.color = LOSING_COLOR
    if game.get_game_state() == 'Win':
        message_label.text = game.get_turn() + " Wins!"
        indexes = game.get_winning_positions()
        for i,j in indexes:  square_buttons[i][j].color = WINNING_COLOR

    game.flip_turn()
